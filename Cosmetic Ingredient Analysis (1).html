# TASK 3: TOKENIZING INGREDIENTS AND CREATING BAG OF WORDS

corpus = []           
ingredient_idx = {}   
idx = 0               

for i, ingredients in enumerate(moisturizers_dry['Ingredients']):
    ingredients_clean = ingredients.lower()
    tokens = ingredients_clean.split(', ')
    corpus.append(tokens)
    for ingredient in tokens:
        if ingredient not in ingredient_idx:
            ingredient_idx[ingredient] = idx
            idx += 1

print(f"Total products processed: {len(corpus)}")
print(f"Unique ingredients found: {len(ingredient_idx)}")

Total products processed: 190
Unique ingredients found: 2233

# TASK 5: CREATING ONE-HOT ENCODER FUNCTION

def oh_encoder(tokens):
    """
    One-hot encoder for ingredient lists
    Takes list of ingredients and returns binary vector
    """
    x = np.zeros(N)
    for ingredient in tokens:
        ingredient_index = ingredient_idx[ingredient]
        x[ingredient_index] = 1
    return x
